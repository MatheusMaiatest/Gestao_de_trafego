-- Análise de campanhas por região e tipo de cliente
SELECT 
    -- Campanha
    fc.campaign_name AS Campanha,
    DATE_FORMAT(fc.metric_date, '%Y-%m') AS Mes,
    
    -- Métricas de Performance
    COUNT(DISTINCT pvt.order_id) AS Total_Pedidos,
    SUM(pvt.quantity) AS Produtos_Vendidos,
    SUM(pvt.price * pvt.quantity) AS Receita_Total,
    SUM(fc.spend) AS Investimento,
    
    -- ROI/ROAS
    (SUM(pvt.price * pvt.quantity) / NULLIF(SUM(fc.spend), 0)) AS ROAS,
    
    -- Análise por Região (Top 5)
    SUM(CASE WHEN ca.state = 'SP' THEN pvt.price * pvt.quantity ELSE 0 END) AS Receita_SP,
    SUM(CASE WHEN ca.state = 'RJ' THEN pvt.price * pvt.quantity ELSE 0 END) AS Receita_RJ,
    SUM(CASE WHEN ca.state = 'MG' THEN pvt.price * pvt.quantity ELSE 0 END) AS Receita_MG,
    
    -- Análise por Tipo de Cliente
    SUM(CASE WHEN LENGTH(tc.cpf) = 11 THEN pvt.price * pvt.quantity ELSE 0 END) AS Receita_PF,
    SUM(CASE WHEN LENGTH(tc.cpf) > 11 THEN pvt.price * pvt.quantity ELSE 0 END) AS Receita_PJ,
    
    -- Clientes Únicos
    COUNT(DISTINCT pet.customer_id) AS Total_Clientes

FROM facebook_campanhas fc

-- Conectar com produtos vendidos (CHAVE!)
INNER JOIN produtos_vendidos_tray_ecommerce pvt 
    ON fc.campaign_id = pvt.id_campaign

-- Conectar com pedidos
INNER JOIN pedidos_ecommerce_tray pet 
    ON pvt.order_id = pet.id

-- Conectar com clientes
LEFT JOIN tray_customers tc 
    ON pet.customer_id = tc.id

-- Conectar com endereços (para UF)
LEFT JOIN tray_customers_customeraddress ca 
    ON tc.id = ca.customer_id

WHERE fc.metric_date >= '2026-01-01'

GROUP BY fc.campaign_name, DATE_FORMAT(fc.metric_date, '%Y-%m')

HAVING Receita_Total > 0

ORDER BY Receita_Total DESC;
